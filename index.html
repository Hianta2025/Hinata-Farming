function sendSMS() { var message = document.getElementById("message").value; var phoneNumber = "+123456789"; // Replace with the actual phone number fetch("https://your-sms-api.com/send", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ to: phoneNumber, message: message }) }) .then(response => response.json()) .then(data => alert("Message sent: " + data.status)) .catch(error => alert("Error sending message: " + error)); }
